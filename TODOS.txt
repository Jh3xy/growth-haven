□ Create Postgres function for wallet credits on like:
  CREATE OR REPLACE FUNCTION process_post_like(
      p_post_id uuid,
      p_liker_id uuid
  ) RETURNS json AS $$
  DECLARE
      v_post_owner_id uuid;
      v_already_liked boolean;
  BEGIN
      -- Check if already liked
      SELECT EXISTS(
          SELECT 1 FROM likes 
          WHERE post_id = p_post_id AND user_id = p_liker_id
      ) INTO v_already_liked;

      IF v_already_liked THEN
          RAISE EXCEPTION 'You have already liked this post';
      END IF;

      -- Get post owner
      SELECT user_id INTO v_post_owner_id
      FROM posts 
      WHERE id = p_post_id;

      IF v_post_owner_id IS NULL THEN
          RAISE EXCEPTION 'Post not found';
      END IF;

      -- Prevent self-liking
      IF v_post_owner_id = p_liker_id THEN
          RAISE EXCEPTION 'You cannot like your own post';
      END IF;

      -- Everything in one transaction
      BEGIN
          -- Insert like
          INSERT INTO likes (post_id, user_id)
          VALUES (p_post_id, p_liker_id);

          -- Credit liker ₦100
          UPDATE members
          SET wallet_balance = wallet_balance + 100
          WHERE id = p_liker_id;

          -- Credit post owner ₦200
          UPDATE members
          SET wallet_balance = wallet_balance + 200
          WHERE id = v_post_owner_id;

          -- Log transactions
          INSERT INTO transactions (user_id, type, label, amount) VALUES
              (p_liker_id, 'blog_like_reward', 'Liked a post', 100),
              (v_post_owner_id, 'blog_post_reward', 'Post was liked', 200);

          RETURN json_build_object(
              'success', true, 
              'liker_earned', 100, 
              'owner_earned', 200
          );

      EXCEPTION WHEN OTHERS THEN
          RAISE EXCEPTION 'Transaction failed: %', SQLERRM;
      END;
  END;
  $$ LANGUAGE plpgsql SECURITY DEFINER;

□ Add has_deposited flag to members table (if not already present):
  ALTER TABLE members ADD COLUMN IF NOT EXISTS has_deposited boolean DEFAULT false;
  
  -- Create trigger to set has_deposited = true on first successful deposit
  CREATE OR REPLACE FUNCTION mark_first_deposit() RETURNS TRIGGER AS $$
  BEGIN
    IF NEW.type = 'deposit' AND NEW.status = 'completed' THEN
      UPDATE members 
      SET has_deposited = true 
      WHERE id = NEW.user_id AND has_deposited = false;
    END IF;
    RETURN NEW;
  END;
  $$ LANGUAGE plpgsql;
  
  CREATE TRIGGER on_deposit_completion
  AFTER INSERT ON transactions
  FOR EACH ROW
  EXECUTE FUNCTION mark_first_deposit();

□ Relative timestamps ("2h ago"):
  function formatRelativeTime(isoString) {
    const now = Date.now();
    const then = new Date(isoString).getTime();
    const diff = now - then;
    const mins = Math.floor(diff / 60000);
    const hours = Math.floor(diff / 3600000);
    const days = Math.floor(diff / 86400000);
    
    if (mins < 60) return `${mins}m ago`;
    if (hours < 24) return `${hours}h ago`;
    if (days < 7) return `${days}d ago`;
    return formatDate(isoString); // fallback to "24 Apr 2026"
  }

□ Create Supabase views for each category:

  -- All-Time Champions
  CREATE VIEW leaderboard_all_time AS
  SELECT 
    m.id,
    m.first_name || ' ' || m.last_name as name,
    m.first_name,
    m.last_name,
    m.wallet_balance as stat,
    m.is_dummy,
    RANK() OVER (ORDER BY m.wallet_balance DESC) as rank
  FROM members m
  ORDER BY rank
  LIMIT 50;
  
  -- Weekly Leaders (earned this week)
  CREATE VIEW leaderboard_weekly AS
  SELECT 
    m.id,
    m.first_name || ' ' || m.last_name as name,
    m.first_name,
    m.last_name,
    COALESCE(SUM(t.amount), 0) as stat,
    m.is_dummy,
    RANK() OVER (ORDER BY COALESCE(SUM(t.amount), 0) DESC) as rank
  FROM members m
  LEFT JOIN transactions t ON t.user_id = m.id 
    AND t.created_at >= date_trunc('week', NOW())
    AND t.type IN ('daily_claim', 'referral_bonus', 'vault_maturity', 'blog_like_reward', 'blog_post_reward')
  GROUP BY m.id
  ORDER BY rank
  LIMIT 50;
  
  -- Investment Pros (completed vault plans)
  CREATE VIEW leaderboard_investments AS
  SELECT 
    m.id,
    m.first_name || ' ' || m.last_name as name,
    m.first_name,
    m.last_name,
    COUNT(i.id) as stat,
    m.is_dummy,
    RANK() OVER (ORDER BY COUNT(i.id) DESC) as rank
  FROM members m
  LEFT JOIN investments i ON i.user_id = m.id AND i.status = 'completed'
  GROUP BY m.id
  ORDER BY rank
  LIMIT 50;
  
  -- Blog Stars (most likes received)
  CREATE VIEW leaderboard_blog_stars AS
  SELECT 
    m.id,
    m.first_name || ' ' || m.last_name as name,
    m.first_name,
    m.last_name,
    COUNT(l.id) as stat,
    m.is_dummy,
    RANK() OVER (ORDER BY COUNT(l.id) DESC) as rank
  FROM members m
  LEFT JOIN posts p ON p.user_id = m.id
  LEFT JOIN likes l ON l.post_id = p.id
  GROUP BY m.id
  ORDER BY rank
  LIMIT 50;
  
  -- Referral Kings
  CREATE VIEW leaderboard_referrals AS
  SELECT 
    m.id,
    m.first_name || ' ' || m.last_name as name,
    m.first_name,
    m.last_name,
    COUNT(r.id) as stat,
    m.is_dummy,
    RANK() OVER (ORDER BY COUNT(r.id) DESC) as rank
  FROM members m
  LEFT JOIN members r ON r.referrer_code = m.referral_code
  GROUP BY m.id
  ORDER BY rank
  LIMIT 50;

□ Create RPC function to get user's rank in a category:
  CREATE OR REPLACE FUNCTION get_user_rank(
    p_category text,
    p_user_id uuid
  ) RETURNS integer AS $$
  DECLARE
    v_rank integer;
  BEGIN
    CASE p_category
      WHEN 'all_time' THEN
        SELECT rank INTO v_rank FROM leaderboard_all_time WHERE id = p_user_id;
      WHEN 'weekly' THEN
        SELECT rank INTO v_rank FROM leaderboard_weekly WHERE id = p_user_id;
      WHEN 'investments' THEN
        SELECT rank INTO v_rank FROM leaderboard_investments WHERE id = p_user_id;
      WHEN 'blog_stars' THEN
        SELECT rank INTO v_rank FROM leaderboard_blog_stars WHERE id = p_user_id;
      WHEN 'referrals' THEN
        SELECT rank INTO v_rank FROM leaderboard_referrals WHERE id = p_user_id;
      ELSE
        RETURN NULL;
    END CASE;
    
    RETURN v_rank;
  END;
  $$ LANGUAGE plpgsql;
